// Docs: https://dbml.dbdiagram.io/docs

Table users {
    id integer [primary key]
    username varchar
    email varchar
    password varchar
    coins integer
    role varchar
    created_at timestamp
    update_at timestamp
}

Table events {
  id integer [primary key]
  event_name varchar
  description text
  active bool
  start_date timestamp
  end_date timestamp
  created_at timestamp
  updated_at timestamp
  images jsonb
  created_by varchar
}

Table gacha_items {
  id integer [primary key]
  event_id int
  item_name varchar
  rarity text
  drop_rate numeric
  description text
  images text
  created_at timestamp
  updated_at timestamp
}

Table user_inventory {
  id integer [primary key]
  user_id int
  item_id int
  qty int
  acquired_at timestamp
  username varchar
}

Table gacha_log{
  id integer [primary key]
  log jsonb
}
